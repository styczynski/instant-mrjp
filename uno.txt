class Point2 {
  int x;
  int y;

  void move (int dx, int dy) {
     this.x = this.x + dx;
     this.y = this.y + dy;
  }

  int getX () { return this.x; }

  int getY () { return this.y; }
}

class Point3 extends Point2 {
  int z;

  void moveZ (int dz) {
    this.z = this.z + dz;
  }

  int getZ () { return this.z; }

}

class Point4 extends Point3 {
  int w;

  void moveW (int dw) {
    this.w = this.w + dw;
  }

  int getW () { return this.w; }

}
